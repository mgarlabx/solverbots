Como usar

// Botão com texto const btnSalvar = Buttons.create({ text: 'Salvar', variant: 'primary', onClick: () => { ... } }); // Botão com ícone + texto const btnApagar = Buttons.create({ text: 'Apagar', icon: Icons.trash, variant: 'danger', size: 'sm', onClick: () => { ... } }); // Botão de ícone com tooltip const btnCopy = Buttons.createIconButton({ icon: Icons.copy, tooltip: 'Copiar', onClick: () => { ... } }); // Toolbar com grupos const toolbar = Buttons.createToolbar({ bordered: true, groups: [ [btnCopy, btnDownload], [btnEdit, btnTrash] ] }); container.appendChild(toolbar); // Processing Buttons.startProcessing(btnSalvar, 'A guardar...'); // ... quando terminar: Buttons.stopProcessing(btnSalvar);
